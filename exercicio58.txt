
# Verificação se o pip está instalado:
pip --version

# Instalação do pacote requests:
pip install requests

# Instalação de uma versão específica do pacote numpy (versão 1.21.0):
pip install numpy==1.21.0

# Atualização do pacote requests para a versão mais recente:
pip install --upgrade requests

# Remoção do pacote requests:
pip uninstall -y requests

# Listagem de todos os pacotes instalados:
pip list
